/** * 이노시뮬레이션 명함 신청 처리 스크립트 * * [설정 방법] * 1. script.google.com 접속 → 새 프로젝트 * 2. 이 코드를 그대로 붙여넣기 * 3. 아래 SHEET_ID 값을 본인의 구글시트 ID로 교체 * (구글시트 주소 https://docs.google.com/spreadsheets/d/여기부분/edit 중 "여기부분") * 4. 오른쪽 상단 [배포] → [새 배포] → 유형: 웹 앱 * - 실행 대상: 나(본인) * - 액세스 권한: 모든 사용자 * 5. 배포 후 나오는 웹 앱 URL을 복사해서 index.html의 APPS_SCRIPT_URL 자리에 붙여넣기 */ const RECIPIENT_EMAIL = "syj@innosim.com"; const SHEET_ID = "1Lzv-oOiPZl3ZvimTa5Dlk6MvGGfVAMdYS5UwUBt6Z1k"; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const { name, englishName, dept, deptEn, rank, rankEn, mobile, email, front, back } = data; const frontBlob = dataUrlToBlob(front, `${name}_명함_앞면.png`); const backBlob = dataUrlToBlob(back, `${name}_명함_뒤면.png`); const docxBlob = buildDocxBlob({ name, englishName, dept, deptEn, rank, rankEn, mobile, email }); MailApp.sendEmail({ to: RECIPIENT_EMAIL, subject: `[명함 신청] ${name} (${dept})`, body: `명함 신청이 접수되었습니다.\n\n` + `이름: ${name}\n` + `영문명: ${englishName}\n` + `부서: ${dept}\n` + `직급: ${rank}\n` + `전화번호: ${mobile}\n` + `이메일: ${email}\n\n` + `첨부된 명함 이미지(앞/뒤)와 워드 신청서를 확인해 주세요. 워드 파일은 그대로 업체에 전달하시면 됩니다.`, attachments: [frontBlob, backBlob, docxBlob], }); if (SHEET_ID && SHEET_ID.indexOf("여기에") !== 0) { const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet(); sheet.appendRow([new Date(), name, englishName, dept, deptEn, rank, rankEn, mobile, email]); } return ContentService .createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: String(err) })) .setMimeType(ContentService.MimeType.JSON); } } function dataUrlToBlob(dataUrl, filename) { const base64 = dataUrl.split(",")[1]; const bytes = Utilities.base64Decode(base64); return Utilities.newBlob(bytes, "image/png", filename); } /** * 신청 정보를 워드(.docx) 파일로 만들어 반환합니다. * 구글 문서를 임시로 만든 뒤 .docx로 변환하고, 원본 구글 문서는 휴지통으로 보냅니다. */ function buildDocxBlob(data) { const tempName = `명함신청서_${data.name}_${new Date().getTime()}`; const doc = DocumentApp.create(tempName); const body = doc.getBody(); body.appendParagraph("명함 신청서").setHeading(DocumentApp.ParagraphHeading.HEADING1); body.appendParagraph(" "); const rows = [ ["이름", data.name], ["영문명", data.englishName || ""], ["부서", data.dept || ""], ["부서(영문)", data.deptEn || ""], ["직급", data.rank || ""], ["직급(영문)", data.rankEn || ""], ["전화번호", data.mobile || ""], ["이메일", (data.email || "") + "@innosim.com"], ]; const table = body.appendTable(rows); for (let i = 0; i < rows.length; i++) { table.getRow(i).getCell(0).setBold(true); } doc.saveAndClose(); const file = DriveApp.getFileById(doc.getId()); const docxBlob = file.getAs("application/vnd.openxmlformats-officedocument.wordprocessingml.document"); docxBlob.setName(`${data.name}_명함신청서.docx`); file.setTrashed(true); // 임시로 만든 구글 문서는 휴지통으로 이동(정리) return docxBlob; }
